title: "Quickstart"
description: "Upload an image to Cloudinary, apply a transformation, and display it in a UIImageView — all in under 10 minutes."
---

This guide walks you through the minimum steps to get a working media pipeline in your iOS app: configure the SDK, upload a file, generate a transformed delivery URL, and display the result.

<Note>
  You'll need a free Cloudinary account. Your **cloud name** and **API key** are in your [Cloudinary Console](https://cloudinary.com/console) under **Account Details**.
</Note>

<Steps>
  <Step title="Install the SDK">
    Add the Cloudinary iOS SDK to your project. The fastest path is CocoaPods — add the pod to your `Podfile` and run `pod install`:

    ```ruby Podfile
    source 'https://github.com/CocoaPods/Specs.git'
    platform :ios, '9.0'
    use_frameworks!

    target 'MyApp' do
      pod 'Cloudinary', '~> 5.0'
    end
    ```

    ```bash
    pod install
    ```

    For Carthage, Swift Package Manager, or manual setup, see [Installation](/installation).
  </Step>

  <Step title="Configure your credentials">
    Create a `CLDConfiguration` with your cloud name and API key, then initialize `CLDCloudinary`. The `CLDCloudinary` instance is the entry point for all SDK operations — keep a reference to it (for example, as a property on your app delegate or a shared service).

    ```swift
    import Cloudinary

    let config = CLDConfiguration(cloudName: "YOUR_CLOUD_NAME", apiKey: "YOUR_API_KEY")
    let cloudinary = CLDCloudinary(configuration: config)
    ```

    If you'd rather not hard-code credentials, you can pass them as a Cloudinary URL instead:

    ```swift
    // Using a cloudinary:// URL
    let config = CLDConfiguration(cloudinaryUrl: "cloudinary://YOUR_API_KEY:YOUR_API_SECRET@YOUR_CLOUD_NAME")
    let cloudinary = CLDCloudinary(configuration: config)
    ```

    Or read from the `CLOUDINARY_URL` environment variable at launch:

    ```swift
    // Reads the CLOUDINARY_URL environment variable
    if let config = CLDConfiguration.initWithEnvParams() {
        let cloudinary = CLDCloudinary(configuration: config)
    }
    ```

    <Warning>
      Never ship your **API secret** inside a production app binary. Use unsigned uploads with upload presets for client-side flows, and keep signing logic on a backend. See [Unsigned upload](/guides/unsigned-upload).
    </Warning>
  </Step>

  <Step title="Upload a file">
    Use `CLDUploader` to upload a local file. Call `createUploader()` on your `CLDCloudinary` instance and pass a `URL` pointing to the file. The SDK assigns a random public ID and returns the result in the completion handler.

    ```swift
    guard let fileUrl = Bundle.main.url(forResource: "sample", withExtension: "jpg") else { return }

    cloudinary.createUploader().upload(
        file: fileUrl,
        params: nil,
        progress: { bytesUploaded, totalBytes, totalBytesExpected in
            // Update a progress indicator
            let progress = Double(bytesUploaded) / Double(totalBytesExpected)
            print("Upload progress: \(progress)")
        },
        completionHandler: { response, error in
            if let error = error {
                print("Upload failed: \(error.localizedDescription)")
                return
            }
            if let publicId = response?.publicId {
                print("Uploaded successfully. Public ID: \(publicId)")
            }
        }
    )
    ```

    The `upload` method returns a `CLDUploadRequest`, which you can call `.cancel()`, `.suspend()`, or `.resume()` on at any time.

    <Tip>
      To set a custom public ID or apply an incoming transformation before saving, pass a `CLDUploadRequestParams` instance. See [Upload basics](/guides/upload-basics) for the full parameter reference.
    </Tip>
  </Step>

  <Step title="Generate a transformed URL">
    Use `CLDUrl` to build a Cloudinary delivery URL. Chain `setTransformation()` before calling `generate()` to encode transformation parameters directly into the URL — no separate API call is needed.

    ```swift
    // Plain URL for an uploaded image
    let plainUrl = cloudinary.createUrl().generate("sample.jpg")
    // https://res.cloudinary.com/YOUR_CLOUD_NAME/image/upload/sample.jpg

    // 100×150 crop
    let transformation = CLDTransformation()
        .setWidth(100)
        .setHeight(150)
        .setCrop(.crop)

    let croppedUrl = cloudinary.createUrl()
        .setTransformation(transformation)
        .generate("sample.jpg")
    // https://res.cloudinary.com/YOUR_CLOUD_NAME/image/upload/c_crop,h_150,w_100/sample.jpg

    // 90×90 face-detection thumbnail
    let thumbTransformation = CLDTransformation()
        .setWidth(90)
        .setHeight(90)
        .setCrop(.Thumb)
        .setGravity(.Face)

    let thumbUrl = cloudinary.createUrl()
        .setTransformation(thumbTransformation)
        .generate("sample.jpg")
    // https://res.cloudinary.com/YOUR_CLOUD_NAME/image/upload/c_thumb,g_face,h_90,w_90/sample.jpg
    ```

    `generate()` returns an optional `String?`. The URL is `nil` only if a required configuration value is missing or the transformation is invalid.
  </Step>

  <Step title="Display the image">
    Pass the generated URL string to `CLDDownloader` to fetch and display the image. The downloader checks the in-memory cache first (default 30 MB), then the disk cache (default 150 MB), and only downloads from the network when needed.

    ```swift
    guard let imageUrl = cloudinary.createUrl()
        .setTransformation(CLDTransformation().setWidth(300).setHeight(300).setCrop(.fill))
        .generate("sample.jpg") else { return }

    cloudinary.createDownloader().fetchImage(
        imageUrl,
        { bytesDownloaded, totalBytes, totalBytesExpected in
            // Handle download progress
        },
        completionHandler: { responseImage, error in
            DispatchQueue.main.async {
                if let image = responseImage {
                    self.imageView.image = image
                }
            }
        }
    )
    ```

    If you're working with `UIImageView`, the SDK's built-in extension makes this even shorter:

    ```swift
    // Set image directly on a UIImageView using its public ID
    imageView.cldSetImage(
        publicId: "sample",
        cloudinary: cloudinary,
        transformation: CLDTransformation().setWidth(300).setHeight(300).setCrop(.fill)
    )
    ```
  </Step>
</Steps>

## Complete example

The snippet below brings all five steps together in a single `UIViewController`.

```swift
import UIKit
import Cloudinary

class MediaViewController: UIViewController {

    @IBOutlet weak var imageView: UIImageView!

    // Keep a strong reference to the CLDCloudinary instance
    let cloudinary: CLDCloudinary = {
        let config = CLDConfiguration(cloudName: "YOUR_CLOUD_NAME", apiKey: "YOUR_API_KEY")
        return CLDCloudinary(configuration: config)
    }()

    override func viewDidLoad() {
        super.viewDidLoad()
        uploadAndDisplay()
    }

    func uploadAndDisplay() {
        guard let fileUrl = Bundle.main.url(forResource: "sample", withExtension: "jpg") else {
            return
        }

        // 1. Upload the file
        cloudinary.createUploader().upload(
            file: fileUrl,
            params: nil,
            completionHandler: { [weak self] response, error in
                guard let self = self, let publicId = response?.publicId, error == nil else {
                    print("Upload error: \(error?.localizedDescription ?? "unknown")")
                    return
                }

                // 2. Build a 300×300 fill-crop delivery URL
                let transformation = CLDTransformation()
                    .setWidth(300)
                    .setHeight(300)
                    .setCrop(.fill)

                guard let imageUrl = self.cloudinary.createUrl()
                    .setTransformation(transformation)
                    .generate(publicId) else { return }

                // 3. Fetch and display the image
                self.cloudinary.createDownloader().fetchImage(
                    imageUrl,
                    completionHandler: { responseImage, fetchError in
                        DispatchQueue.main.async {
                            if let image = responseImage {
                                self.imageView.image = image
                            }
                        }
                    }
                )
            }
        )
    }
}
```
